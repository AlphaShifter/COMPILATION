int main()
{
    int start := 2;
    int end := 100;
    int p := start;

    int flag := 0;

    while(p < end+1)
    {
        int i := 2;
        int j := 2;

        while (i<p)
        {
            while (j<p)
            {
                if (i*j = p)
                {
                     flag := 1;
                }
                j := j+1;
            }
            i := i+1;
        }

        if (flag = 0)
        {
            PrintInt(p);
        }
        flag := 0;
    }
}
