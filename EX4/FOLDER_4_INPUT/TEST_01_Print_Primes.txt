int IsPrime(int p)
{
    int i := 2;
    int j := 2;
    
    while (i<p)
    {
        j := 2;
        while (j<p)
        {
            if (i*j = p)
            {
                 return 0;
            }
            j := j+1;
        }
        i := i+1;
    }
    return 1;
}

void PrintPrimes(int start, int end)
{
    int p := start;
    string s := "TT";
    
    while (p < end+1)
    {
        if (IsPrime(p))
        {
            PrintInt(p);
            PrintString(s);
        }
        p := p + 1;
    }
}

void main()
{
    PrintPrimes(2,100);
}
